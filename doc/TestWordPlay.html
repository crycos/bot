Testing Best Sentence Choice Test that the correct sentence is chosen, given the input

# File test_wordplay.rb, line 24
def test_sentence_choice
  assert_equal('This is a great test',
    WordPlay.best_sentence(['This is a test',
                          'This is another test',
                          'This is a great test'],
                        %{test great this}))
  assert_equal('This is a great test',
    WordPlay.best_sentence(['This is a great test'],
                        %{still the best}))
end
